# 가상환경 키기
. venv/Scripts/activate

# 폴더 이동
cd IOT

# DB 생성(한 번만 해주면 됨)
python manage.py makemigrations
python manage.py migrate

# 서버 구동
python manage.py runserver

# 링크 클릭
아래에서 두 번째 줄에 있는 "http://127.0.0.1:8000/" 를 "Ctrl + 마우스클릭" 하면 됨

######################################################################################################

# IoT 기기 연동
1. 모디 초음파 센서 USB 연결
2. 터미널 하나 더 만들기: 우측 하단 휴지통 아이콘 바로 왼쪽에 있는 분할 아이콘 클릭
3. 폴더 이동: cd IOT
4. 터미널에 명령어 입력: python congression.py
